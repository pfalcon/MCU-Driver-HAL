30 /**
31  * \defgroup hal_us_ticker Microsecond Ticker
32  * Low level interface to the microsecond ticker of a target
33  *
34  * # Defined behavior
35  * * Has a reported frequency between 250KHz and 8MHz for counters which are less than 32 bits wide - Verified by test ::us_ticker_info_test
36  * * Has a reported frequency up to 100MHz for counters which are 32 bits wide - Verified by test ::us_ticker_info_test
37  * * Has a counter that is at least 16 bits wide - Verified by test ::us_ticker_info_test
38  * * All behavior defined by the @ref hal_ticker_shared "ticker specification"
39  *
40  * # Undefined behavior
41  * * See the @ref hal_ticker_shared "ticker specification"
42  *
43  * @see hal_us_ticker_tests
44  *
45  * # Compile-time optimization macros
46  *
47  * To permit compile-time optimization, particularly of wait_us, the following macros should
48  * be defined by a target's device.h:
49  *
50  * US_TICKER_PERIOD_NUM, US_TICKER_PERIOD_DEN: These denote the ratio (numerator, denominator)
51  * of the ticker period to a microsecond. For example, an 8MHz ticker would have NUM = 1, DEN = 8;
52  * a 1MHz ticker would have NUM = 1, DEN = 1; a 250kHz ticker would have NUM = 4, DEN = 1.
53  * Both numerator and denominator must be 16 bits or less, but need not be fully simplified.
54  *
55  * US_TICKER_MASK: The value mask for the ticker - eg 0x07FFFFFF for a 27-bit ticker.
56  *
57  * If any are defined, all 3 must be defined, and the macros are checked for consistency with
58  * us_ticker_get_info by test ::us_ticker_info_test.
59 
60  * @{
61  */

64 /**
65  * \defgroup hal_ticker_shared Ticker Hal
66  * Low level interface to the ticker of a target
67  *
68  * # Defined behavior
69  * * The function ticker_init is safe to call repeatedly - Verified by test ::ticker_init_test
70  * * The function ticker_init allows the ticker to keep counting and disables the ticker interrupt - Verified by test ::ticker_init_test
71  * * Ticker frequency is non-zero and counter is at least 8 bits - Verified by ::ticker_info_test
72  * * The ticker rolls over at (1 << bits) and continues counting starting from 0 - Verified by ::ticker_overflow_test
73  * * The ticker counts at the specified frequency +- 10% - Verified by ::ticker_frequency_test
74  * * The ticker increments by 1 each tick - Verified by ::ticker_increment_test
75  * * The ticker interrupt fires only when the ticker times increments to or past the value set by ticker_set_interrupt.
76  * Verified by ::ticker_interrupt_test and ::ticker_past_test
77  * * It is safe to call ticker_set_interrupt repeatedly before the handler is called - Verified by ::ticker_repeat_reschedule_test
78  * * The function ticker_fire_interrupt causes ticker_irq_handler to be called immediately from interrupt context -
79  * Verified by ::ticker_fire_now_test
80  * * The ticker operations ticker_read, ticker_clear_interrupt, ticker_set_interrupt and ticker_fire_interrupt
81  * take less than 20us to complete - Verified by ::ticker_speed_test
82  * * The ticker operations ticker_init and ticker_read are atomic.
83  *
84  * # Undefined behavior
85  * * Calling any function other than ticker_init before the initialization of the ticker
86  * * Whether ticker_irq_handler is called a second time if the time wraps and matches the value set by ticker_set_interrupt again
87  * * Calling ticker_set_interrupt with a value that has more than the supported number of bits
88  * * Calling any function other than us_ticker_init after calling us_ticker_free
89  *
90  * # Potential bugs
91  * * Drift due to reschedule - Verified by ::ticker_repeat_reschedule_test
92  * * Incorrect overflow handling of timers - Verified by ::ticker_overflow_test
93  * * Interrupting at a time of 0 - Verified by ::ticker_overflow_test
94  * * Interrupt triggered more than once - Verified by ::ticker_interrupt_test
95  *
96  * @ingroup hal_us_ticker
97  * @ingroup hal_lp_ticker
98  */
99

126 /** Initialize the ticker
127  *
128  * Initialize or re-initialize the ticker. This resets all the
129  * clocking and prescaler registers, along with disabling
130  * the compare interrupt.
131  *
132  * @note Initialization properties tested by ::ticker_init_test
133  *
134  * Pseudo Code:
135  * @code
136  * void us_ticker_init()
137  * {
138  * // Enable clock gate so processor can read TIMER registers
139  * POWER_CTRL |= POWER_CTRL_TIMER_Msk;
140  *
141  * // Disable the timer and ensure it is powered down
142  * TIMER_CTRL &= ~(TIMER_CTRL_ENABLE_Msk | TIMER_CTRL_COMPARE_ENABLE_Msk);
143  *
144  * // Configure divisors
145  * uint32_t prescale = SystemCoreClock / 1000000;
146  * TIMER_PRESCALE = prescale - 1;
147  * TIMER_CTRL |= TIMER_CTRL_ENABLE_Msk;
148  *
149  * // Install the interrupt handler
150  * NVIC_SetVector(TIMER_IRQn, (uint32_t)us_ticker_irq_handler);
151  * NVIC_EnableIRQ(TIMER_IRQn);
152  * }
153  * @endcode
154  */
155 void us_ticker_init(void);
156 
157 /** Deinitialize the us ticker
158  *
159  * Powerdown the us ticker in preparation for sleep, powerdown, or reset.
160  *
161  * After this function is called, no other ticker functions should be called
162  * except us_ticker_init(), calling any function other than init is undefined.
163  *
164  * @note This function stops the ticker from counting.
165  *
166  * Pseudo Code:
167  * @code
168  * uint32_t us_ticker_free()
169  * {
170  * // Disable timer
171  * TIMER_CTRL &= ~TIMER_CTRL_ENABLE_Msk;
172  *
173  * // Disable the compare interrupt
174  * TIMER_CTRL &= ~TIMER_CTRL_COMPARE_ENABLE_Msk;
175  *
176  * // Disable timer interrupt
177  * NVIC_DisableIRQ(TIMER_IRQn);
178  *
179  * // Disable clock gate so processor cannot read TIMER registers
180  * POWER_CTRL &= ~POWER_CTRL_TIMER_Msk;
181  * }
182  * @endcode
183  *
184  */
185 void us_ticker_free(void);
