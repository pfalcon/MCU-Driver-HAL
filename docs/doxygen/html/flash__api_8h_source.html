55 /** Erase one sector starting at defined address
56  *
57  * The address should be at sector boundary. This function does not do any check for address alignments
58  * @param obj The flash object
59  * @param address The sector starting address
60  * @return 0 for success, -1 for error
61  */
62 int32_t flash_erase_sector(flash_t *obj, uint32_t address);
63 
64 /** Read data starting at defined address
65  *
66  * This function has a WEAK implementation using memcpy for backwards compatibility.
67  * @param obj The flash object
68  * @param address Address to begin reading from
69  * @param data The buffer to read data into
70  * @param size The number of bytes to read
71  * @return 0 for success, -1 for error
72  */
73 int32_t flash_read(flash_t *obj, uint32_t address, uint8_t *data, uint32_t size);
74 
75 /** Program pages starting at defined address
76  *
77  * The pages should not cross multiple sectors.
78  * This function does not do any check for address alignments or if size is aligned to a page size.
79  * @param obj The flash object
80  * @param address The sector starting address
81  * @param data The data buffer to be programmed
82  * @param size The number of bytes to program
83  * @return 0 for success, -1 for error
84  */
85 int32_t flash_program_page(flash_t *obj, uint32_t address, const uint8_t *data, uint32_t size);
86 
87 /** Get sector size
88  *
89  * @param obj The flash object
90  * @param address The sector starting address
91  * @return The size of a sector
92  */
93 uint32_t flash_get_sector_size(const flash_t *obj, uint32_t address);
94 
95 /** Get page size
96  *
97  * The page size defines the writable page size
98  * @param obj The flash object
99  * @return The size of a page
100  */
101 uint32_t flash_get_page_size(const flash_t *obj);
102

110 /** Get the flash region size

115 uint32_t flash_get_size(const flash_t *obj);
