31 /**
32  * \defgroup boot Boot sequence
33  * Boot sequence overview
34  *
35  * 1. Setup target
36  * - Configure clocks
37  * - Configure watchdog (if applicable)
38  * - Turn on RAM (if applicable)
39  * - Jump to setup toolchain
40  * 2. Setup toolchain
41  * - Initialize RAM
42  * - Initialize standard library
43  * - Call mbed_init
44  * - jump to start rtos
45  * 3. Start RTOS
46  * - Create main thread
47  * - Start scheduler
48  * - main thread calls start mbed
49  * 4. Start mbed
50  * - Call mbed_tfm_init
51  * - Call mbed_main
52  * - Call main
53  *
54  * @{
55  */
56 
57 /* Heap limits - only used if set */
58 extern unsigned char *mbed_heap_start;
59 extern uint32_t mbed_heap_size;
60 
61 /* Stack limits */
62 extern unsigned char *mbed_stack_isr_start;
63 extern uint32_t mbed_stack_isr_size;

65 /**
66  * Perform low level init of mbed
67  *
68  * The toolchain calls this function as part of the boot
69  * sequence. This function does the following:
70  * - Sets up NVIC so interrupts can be used
71  * - Calls into the vendor SDK so it can be used
72  * - Initializes the RTOS so it can be used
73  *
74  * Preconditions:
75  * - Target is initialized
76  * - Ram is initialized
77  * - value set for mbed_heap_start
78  * - value set for mbed_heap_size
79  * - value set for mbed_stack_isr_start
80  * - value set for mbed_stack_isr_size
81  *
82  * The following events must not have happened yet:
83  * - global constructors must not be called yet
84  *
85  */
86 void mbed_init(void);
87 
88 /**
89  * Start the main mbed application
90  *
91  * This is the last stage of the boot sequence. This function must be
92  * called only after the RTOS has been fully initialized.
93  * This function does the following:
94  * - Initialize the toolchain
95  * - Run mbed_main
96  * - Run main
97  *
98  * Preconditions:
99  * - The RTOS has been started by a call to mbed_rtos_start
100  *
101  */
102 void mbed_start(void);
103 
104 /**
105  * Perform low level initialization of the RTOS
106  *
107  * Set the RTOS to a known state but don't start the scheduler. After
108  * the RTOS has been initialized it is safe to create RTOS primitives
109  * for file locks or other purposes.
110  *
111  * Preconditions:
112  * - Ram is initialized
113  * - NVIC is setup
114  * - Vendor SDK must be initialized by a call to mbed_sdk_init
115  */
116 void mbed_rtos_init(void);
117 
118 /**
119  * Start the RTOS
120  *
121  * Start the RTOS scheduler and call mbed_start on the
122  * main thread. This function does not return.
123  *
124  * Preconditions:
125  * - RTOS has been initialized by a call to mbed_rtos_init
126  */
127 void mbed_rtos_start(void);
128 
129 /**
130  * Perform toolchain specific initialization
131  *
132  * Initialize locks if this has not been done already
133  * and call global C++ constructors.
134  *
135  * Preconditions:
136  * - The RTOS has been started by a call to mbed_rtos_start
137  */
138 void mbed_toolchain_init(void);
139 
140 /**
141  * SDK hook for running code before ctors or OS
142  *
143  * This is a weak function which can be overridden by a target's
144  * SDK to allow code to run after ram is initialized but before
145  * the OS has been started or constructors have run.
146  *
147  * Preconditions:
148  * - Ram is initialized
149  * - NVIC is setup
150  */
151 void mbed_sdk_init(void);
152 
153 /**
154  * TF-M specific application hook for running code before mbed_main
155  *
156  * This is a weak function which can be overridden by an application
157  * to allow code to run before mbed_main is called.
158  * Some TF-M supported platforms require synchronization between
159  * Secure and Non-Secure cores, therefore these tasks are done here.
160  *
161  * Preconditions:
162  * - The RTOS has been started by a call to mbed_rtos_start
163  * - The toolchain has been initialized by a call to mbed_toolchain_init
164  */
165 void mbed_tfm_init(void);
166 
167 /**
168  * Application hook for running code before main
169  *
170  * This is a weak function which can be overridden by an application
171  * to allow code to run before main is called.
172  *
173  * Preconditions:
174  * - The RTOS has been started by a call to mbed_rtos_start
175  * - TFM support has been initialized by a call to mbed_tfm_init.
176  * - The toolchain has been initialized by a call to mbed_toolchain_init
177  */
178 void mbed_main(void);
179 
180 /**
181  * Create and Initialize a Singleton Mutex object.
182  *
183  * Precondition(s):
184  * - The RTOS has been started by a call to mbed_rtos_start
185  */
186 void mbed_rtos_init_singleton_mutex(void);
