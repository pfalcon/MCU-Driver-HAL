330 /* Note care taken with types here. Values which the assembler outputs correctly
331  * narrowed, or inputs without caring about width, are marked as type T. Other
332  * values are uint32_t. It's not clear from documentation whether assembler
333  * assumes anything about widths, but try to signal correctly to get necessary
334  * narrowing, and avoid unnecessary.
335  * Tests show that GCC in particular will pass in unnarrowed values - eg passing
336  * "uint8_t arg = -1" to the assembler as 0xFFFFFFFF. This is fine for, eg, add_u8,
337  * but wouldn't be for compare_and_exchange_u8.
338  * On the other hand, it seems to be impossible to stop GCC inserting narrowing
339  * instructions for the output - it will always put in UXTB for the oldValue of
340  * an operation.
341  */
342 #define DO_MBED_LOCKFREE_EXCHG_OP(T, fn_suffix, M) \
343 inline T core_util_atomic_exchange_##fn_suffix(volatile T *valuePtr, T newValue) \
344 { \
345  T oldValue; \
346  uint32_t fail; \
347  MBED_BARRIER(); \
348  DO_MBED_LOCKFREE_EXCHG_ASM(M); \
349  MBED_BARRIER(); \
350  return oldValue; \
351 } \
352  \

354  volatile T *valuePtr, T newValue, mbed_memory_order order) \
355 { \

357  uint32_t fail; \
358  MBED_RELEASE_BARRIER(order); \

360  MBED_ACQUIRE_BARRIER(order); \

362 }

559 // possible unneeded distinction between Thumbv1 and Thumbv2, and

566 // instead of ADDS/SUBS, but then that would block generation
567 // of the 16-bit forms. Shame we can't specify "don't care"
568 // for the "S", or get the GNU multi-alternative to
569 // choose ADDS/ADD appropriately.
