31 /**
32  * \defgroup hal_rtc RTC hal
33  *
34  * The RTC hal provides a low level interface to the Real Time Counter (RTC) of a
35  * target.
36  *
37  * # Defined behaviour
38  * * The function ::rtc_init is safe to call repeatedly - Verified by test ::rtc_init_test.
39  * * RTC accuracy is at least 10% - Verified by test ::rtc_accuracy_test.
40  * * Init/free doesn't stop RTC from counting - Verified by test ::rtc_persist_test.
41  * * Software reset doesn't stop RTC from counting - Verified by test ::rtc_reset_test.
42  * * Sleep modes don't stop RTC from counting - Verified by test ::rtc_sleep_test.
43  * * Shutdown mode doesn't stop RTC from counting - Not verified.
44  * * The functions ::rtc_write/::rtc_read provides availability to set/get RTC time
45  * - Verified by test ::rtc_write_read_test.
46  * * The functions ::rtc_isenabled returns 1 if the RTC is counting and the time has been set,
47  * 0 otherwise - Verified by test ::rtc_enabled_test.
48  *
49  * # Undefined behaviour
50  * * Calling any function other than ::rtc_init before the initialisation of the RTC
51  *
52  * # Potential bugs
53  * * Incorrect overflow handling - Verified by ::rtc_range_test
54  * * Glitches due to ripple counter - Verified by ::rtc_glitch_test
55  *
56  * @{
57  */
58 
59 /** Initialize the RTC peripheral
60  *
61  * Powerup the RTC in perpetration for access. This function must be called
62  * before any other RTC functions ares called. This does not change the state
63  * of the RTC. It just enables access to it.
64  *
65  * @note This function is safe to call repeatedly - Tested by ::rtc_init_test
66  *
67  * Example Implementation Pseudo Code:
68  * @code
69  * void rtc_init()
70  * {
71  * // Enable clock gate so processor can read RTC registers
72  * POWER_CTRL |= POWER_CTRL_RTC_Msk;
73  *
74  * // See if the RTC is already setup
75  * if (!(RTC_STATUS & RTC_STATUS_COUNTING_Msk)) {
76  *
77  * // Setup the RTC clock source
78  * RTC_CTRL |= RTC_CTRL_CLK32_Msk;
79  * }
80  * }
81  * @endcode
82  */
83 void rtc_init(void);
84 
85 /** Deinitialize RTC
86  *
87  * Powerdown the RTC in preparation for sleep, powerdown or reset. That should only
88  * affect the CPU domain and not the time keeping logic.
89  * After this function is called no other RTC functions should be called
90  * except for ::rtc_init.
91  *
92  * @note This function does not stop the RTC from counting - Tested by ::rtc_persist_test
93  *
94  * Example Implementation Pseudo Code:
95  * @code
96  * void rtc_free()
97  * {
98  * // Disable clock gate since processor no longer needs to read RTC registers
99  * POWER_CTRL &= ~POWER_CTRL_RTC_Msk;
100  * }
101  * @endcode
102  */
103 void rtc_free(void);
104 
105 /** Check if the RTC has the time set and is counting
106  *
107  * @retval 0 The time reported by the RTC is not valid
108  * @retval 1 The time has been set the RTC is counting
109  *
110  * Example Implementation Pseudo Code:
111  * @code
112  * int rtc_isenabled()
113  * {
114  * if (RTC_STATUS & RTC_STATUS_COUNTING_Msk) {
115  * return 1;
116  * } else {
117  * return 0;
118  * }
119  * }
120  * @endcode
121  */
122 int rtc_isenabled(void);
123 
124 /** Get the current time from the RTC peripheral
125  *
126  * @return The current time in seconds
127  *
128  * @note Some RTCs are not synchronized with the main clock. If
129  * this is the case with your RTC then you must read the RTC time
130  * in a loop to prevent reading the wrong time due to a glitch.
131  * The test ::rtc_glitch_test is intended to catch this bug.
132  *
133  * Example implementation for an unsynchronized ripple counter:
134  * @code
135  * time_t rtc_read()
136  * {
137  * uint32_t val;
138  * uint32_t last_val;
139  *
140  * // Loop until the same value is read twice
141  * val = RTC_SECONDS;
142  * do {
143  * last_val = val;
144  * val = RTC_SECONDS;
145  * } while (last_val != val);
146  *
147  * return (time_t)val;
148  * }
149  * @endcode
150  */
151 time_t rtc_read(void);
152 
153 /** Write the current time in seconds to the RTC peripheral
154  *
155  * @param t The current time to be set in seconds.
156  *
157  * Example Implementation Pseudo Code:
158  * @code
159  * void rtc_write(time_t t)
160  * {
161  * RTC_SECONDS = t;
162  * }
163  * @endcode
164  */
165 void rtc_write(time_t t);
