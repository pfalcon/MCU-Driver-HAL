176 /**
177  * \defgroup hal_gpio GPIO HAL functions
178  *
179  * # Defined behavior
180  * * ::gpio_init and other init functions can be called with NC or a valid PinName for the target - Verified by ::gpio_nc_test
181  * * ::gpio_is_connected can be used to test whether a gpio_t object was initialized with NC - Verified by ::gpio_nc_test
182  * * ::gpio_init initializes the GPIO pin
183  * * ::gpio_free returns the pin owned by the GPIO object to its reset state
184  * * ::gpio_mode sets the mode of the given pin
185  * * ::gpio_dir sets the direction of the given pin
186  * * ::gpio_write sets the gpio output value
187  * * ::gpio_read reads the input value
188  * * ::gpio_init_in inits the input pin and sets mode to PullDefault
189  * * ::gpio_init_in_ex inits the input pin and sets the mode
190  * * ::gpio_init_out inits the pin as an output, with predefined output value 0
191  * * ::gpio_init_out_ex inits the pin as an output and sets the output value
192  * * ::gpio_init_inout inits the pin to be input/output and set pin mode and value
193  * * The GPIO operations ::gpio_write, ::gpio_read take less than 20us to complete
194  * * The function ::gpio_get_capabilities fills the given
195  * `gpio_capabilities_t` instance according to pin capabilities.
196  *
197  * # Undefined behavior
198  * * Calling any ::gpio_mode, ::gpio_dir, ::gpio_write or ::gpio_read on a gpio_t object that was initialized
199  * with NC.
200  * * Calling ::gpio_set with NC.
201  *
202  * @{
203  */
204
