34 /**
35  * Mark the start of a critical section
36  *
37  * This function will be called by core_util_critical_section_enter() each time
38  * the application requests to enter a critical section. The purpose of the
39  * critical section is to ensure mutual-exclusion synchronisation of the
40  * processor by preventing any change in processor control, the default
41  * behaviour requires storing the state of interrupts in the system before
42  * disabling them.
43  *
44  * The critical section code supports nesting. When a thread has entered a
45  * critical section it can make additional calls to
46  * core_util_critical_section_enter() without deadlocking itself. The critical
47  * section driver API tracks the number of nested calls to the critical section.
48  * The critical section will only be exited when
49  * core_util_critical_section_exit() has been called once for each time it
50  * entered the critical section.
51  *
52  * On the first call to enter a critical section this function MUST store the
53  * state of any interrupts or other application settings it will modify to
54  * facilitate the critical section.
55  *
56  * Each successive call to enter the critical section MUST ignore storing or
57  * modifying any application state.
58  *
59  * The default implementation of this function which will save the current state
60  * of interrupts before disabling them. This implementation can be found in
61  * mbed_critical_section_api.c. This behaviour is can be overridden on a per
62  * platform basis by providing a different implementation within the correct
63  * targets directory.
64  */
65 void hal_critical_section_enter(void);
66 
67 
68 /** Mark the end of a critical section.
69  *
70  * The purpose of this function is to restore any state that was modified upon
71  * entering the critical section, allowing other threads or interrupts to change
72  * the processor control.
73  *
74  * This function will be called once by core_util_critical_section_exit() per
75  * critical section on last call to exit. When called, the application MUST
76  * restore the saved interrupt/application state that was saved when entering
77  * the critical section.
78  *
79  * There is a default implementation of this function, it will restore the state
80  * of interrupts that were previously saved when hal_critical_section_enter was
81  * first called, this implementation can be found in
82  * mbed_critical_section_api.c. This behaviour is overridable by providing a
83  * different function implementation within the correct targets directory.
84  */
85 void hal_critical_section_exit(void);
86 
87 
88 /** Determine if the application is currently running in a critical section
89  *
90  * The purpose of this function is to inform the caller whether or not the
91  * application is running in a critical section. This is done by checking if
92  * the current interrupt state has been saved in the underlying implementation,
93  * this could also be done by checking the state of the interrupts at the time
94  * of calling.
95  *
96  * @return True if running in a critical section, false if not.
97  */
98 bool hal_in_critical_section(void);
