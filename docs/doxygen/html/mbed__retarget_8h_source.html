98 /** Targets may implement this to change stdin, stdout, stderr.
99  *
100  * If the application hasn't provided mbed_override_console, this is called
101  * to give the target a chance to specify a FileHandle for the console.
102  *
103  * If this is not provided or returns NULL, the console will be:
104  * - BufferedSerial if configuration option "platform.stdio-buffered-serial" is
105  * true and the target has DEVICE_SERIAL;
106  * - Raw HAL serial via serial_getc and serial_putc if
107  * "platform.stdio-buffered-serial" is false and the target has DEVICE_SERIAL;
108  * - stdout/stderr will be a sink and stdin will input a stream of 0s if the
109  * target does not have DEVICE_SERIAL.
110  *
111  * @param fd file descriptor - STDIN_FILENO, STDOUT_FILENO or STDERR_FILENO
112  * @return pointer to FileHandle to override normal stream otherwise NULL
113  */
114 FileHandle *mbed_target_override_console(int fd);

116 /** Applications may implement this to change stdin, stdout, stderr.
117  *
118  * This hook gives the application a chance to specify a custom FileHandle
119  * for the console.
120  *
121  * If this is not provided or returns NULL, the console will be specified
122  * by mbed_target_override_console, else will default to serial - see
123  * mbed_target_override_console for more details.
124  *
125  * Example using BufferedSerial:
126  * @code
127  * FileHandle *mbed::mbed_override_console(int) {
128  * static BufferedSerial my_serial(D0, D1);
129  * return &my_serial;
130  * }
131  * @endcode
132  *
133  * Example using SingleWireOutput:
134  * @code
135  * FileHandle *mbed::mbed_override_console(int) {
136  * static SerialWireOutput swo;
137  * return &swo;
138  * }
139  * @endcode
140  *
141  * Example using arm semihosting:
142  * @code
143  * FileHandle *mbed::mbed_override_console(int fileno) {
144  * static LocalFileSystem fs("host");
145  * if (fileno == STDIN_FILENO) {
146  * static FileHandle *in_terminal;
147  * static int in_open_result = fs.open(&in_terminal, ":tt", O_RDONLY);
148  * return in_terminal;
149  * } else {
150  * static FileHandle *out_terminal;
151  * static int out_open_result = fs.open(&out_terminal, ":tt", O_WRONLY);
152  * return out_terminal;
153  * }
154  * }
155  * @endcode
156  *
157  * @param fd file descriptor - STDIN_FILENO, STDOUT_FILENO or STDERR_FILENO
158  * @return pointer to FileHandle to override normal stream otherwise NULL
159  */
160 FileHandle *mbed_override_console(int fd);
