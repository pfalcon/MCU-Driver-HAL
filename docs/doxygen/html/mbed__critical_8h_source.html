33 /** Determine the current interrupts enabled state
34  *
35  * This function can be called to determine whether or not interrupts are currently enabled.
36  * @note
37  * NOTE:
38  * This function works for both cortex-A and cortex-M, although the underlying implementation
39  * differs.
40  * @return true if interrupts are enabled, false otherwise
41  */
42 bool core_util_are_interrupts_enabled(void);
43 
44 /** Determine if this code is executing from an interrupt
45  *
46  * This function can be called to determine if the code is running on interrupt context.
47  * @note
48  * NOTE:
49  * This function works for both cortex-A and cortex-M, although the underlying implementation
50  * differs.
51  * @return true if in an isr, false otherwise
52  */
53 bool core_util_is_isr_active(void);
54 
55 /** Mark the start of a critical section
56  *
57  * This function should be called to mark the start of a critical section of code.
58  * @note
59  * NOTES:
60  * 1) The use of this style of critical section is targetted at C based implementations.
61  * 2) These critical sections can be nested.
62  * 3) The interrupt enable state on entry to the first critical section (of a nested set, or single
63  * section) will be preserved on exit from the section.
64  * 4) This implementation will currently only work on code running in privileged mode.
65  */
66 void core_util_critical_section_enter(void);
67 
68 /** Mark the end of a critical section
69  *
70  * This function should be called to mark the end of a critical section of code.
71  * @note
72  * NOTES:
73  * 1) The use of this style of critical section is targetted at C based implementations.
74  * 2) These critical sections can be nested.
75  * 3) The interrupt enable state on entry to the first critical section (of a nested set, or single
76  * section) will be preserved on exit from the section.
77  * 4) This implementation will currently only work on code running in privileged mode.
78  */
79 void core_util_critical_section_exit(void);
80 
81 /**
82  * Determine if we are currently in a critical section
83  *
84  * @return true if in a critical section, false otherwise.
85  */
86 bool core_util_in_critical_section(void);
