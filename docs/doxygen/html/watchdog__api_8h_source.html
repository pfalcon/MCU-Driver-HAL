28 /**
29  * \defgroup hal_watchdog Watchdog HAL API
30  * Low-level interface to the Independent Watchdog Timer of a target.
31  *
32  * This module provides platform independent access to the system watchdog timer
33  * which is an embedded peripheral that will reset the system in the case of
34  * system failures or malfunctions.
35  *
36  * The watchdog timer initializes a system timer with a time period specified in
37  * the configuration. This timer counts down and triggers a system reset when it
38  * wraps. To prevent the system reset the timer must be continually
39  * kicked/refreshed by calling ::hal_watchdog_kick which will reset the countdown
40  * to the user specified reset value.
41  *
42  * # Defined behavior
43  * * Sleep and debug modes don't stop the watchdog timer from counting down.
44  * * The function ::hal_watchdog_init is safe to call repeatedly. The
45  * function's implementation must not do anything if ::hal_watchdog_init has
46  * already initialized the hardware watchdog timer.
47  * * Maximum supported timeout is `UINT32_MAX` milliseconds; minimum timeout
48  * is 1 millisecond.
49  * * The uncalibrated watchdog should trigger at or after the timeout value
50  * multiplied by the frequency accuracy ratio of its oscillator (typical_frequency / max_frequency).
51  * * The calibrated watchdog should trigger at or after the timeout value.
52  * * The watchdog should trigger before twice the timeout value.
53  *
54  * # Undefined behavior
55  * * Calling any function other than ::hal_watchdog_init or
56  * ::hal_watchdog_get_platform_features before you have initialized the watchdog.
57  *
58  * # Notes
59  * * A software reset may not stop the watchdog timer; the behavior is platform specific.
60  *
61  * @{
62  */
63 
64 /** Watchdog configuration.
65  */
66 typedef struct {
67  /**
68  * Refresh value for the watchdog in milliseconds. The maximum value of this
69  * setting is platform dependent, to find the maximum value for the current
70  * platform call hal_watchdog_get_features() and check the timeout value
71  * member. The minimum valid value for this setting is 1. Attempting to
72  * initialize the watchdog with a timeout of 0 ms returns
73  * WATCHDOG_STATUS_INVALID_ARGUMENT.
74  */
75  uint32_t timeout_ms;
76 } watchdog_config_t;

116 /** Initialize and start a watchdog timer with the given configuration.
117  *
118  * If the watchdog timer is configured and starts successfully, this
119  * function returns ::WATCHDOG_STATUS_OK.
120  *
121  * If the timeout specified is outside the range supported by the platform,
122  * it returns ::WATCHDOG_STATUS_INVALID_ARGUMENT.
123  *
124  * @param[in] config Configuration settings for the watchdog timer
125  *
126  * @return ::WATCHDOG_STATUS_OK if the watchdog is configured correctly and
127  * has started. Otherwise a status indicating the fault.
128  */
129 watchdog_status_t hal_watchdog_init(const watchdog_config_t *config);
130 
131 /** Refreshes the watchdog timer.
132  *
133  * Call this function periodically before the watchdog times out.
134  * Otherwise, the system resets.
135  *
136  * If a watchdog is not running, this function does nothing.
137  */
138 void hal_watchdog_kick(void);
139 
140 /** Stops the watchdog timer.
141  *
142  * Calling this function disables any running watchdog
143  * timers if the current platform supports them.
144  *
145  * @return Returns ::WATCHDOG_STATUS_OK if the watchdog timer was succesfully
146  * stopped, or if the timer was never started. Returns
147  * ::WATCHDOG_STATUS_NOT_SUPPORTED if the watchdog cannot be disabled on
148  * the current platform.
149  */
150 watchdog_status_t hal_watchdog_stop(void);
151 
152 /** Get the watchdog timer refresh value.
153  *
154  * This function returns the configured refresh timeout of the watchdog timer.
155  *
156  * @return Reload value for the watchdog timer in milliseconds.
157  */
158 uint32_t hal_watchdog_get_reload_value(void);
159 
160 /** Get information on the current platforms supported watchdog functionality.
161  *
162  * @return watchdog_feature_t indicating supported watchdog features on the
163  * current platform
164  */
165 watchdog_features_t hal_watchdog_get_platform_features(void);
