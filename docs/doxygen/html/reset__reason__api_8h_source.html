31 /**
32  * \defgroup hal_reset_reason ResetReason HAL API
33  * Low-level interface to the ResetReason of a target.
34  *
35  * This module provides a platform-independent method of checking the cause
36  * of the last system reset.
37  *
38  * # Defined behavior
39  * * The function ::hal_reset_reason_clear clears the ResetReason registers
40  * for the next system boot.
41  * * By the time the user calls ::hal_reset_reason_get to read the value,
42  * some other part of the application may have cleared the value. Therefore,
43  * though there may have been a reset reason in the registers when the system
44  * started, the reason may not be available when the user comes to check it.
45  * * The function ::hal_reset_reason_get_capabilities fills the given
46  * `reset_reason_capabilities_t` instance.
47  *
48  * # Undefined behavior
49  * * There is no guarantee that the function ::hal_reset_reason_get will
50  * return the same value when you call it repeatedly. Store the value for later
51  * use instead of calling the function repeatedly.
52  * * The function ::hal_reset_reason_clear may not clear the ResetReason
53  * register in time for the next system boot.
54  *
55  * # Notes
56  * * The contents of the targets ResetReason register may be cleared by some
57  * subsystem before it first gets called. This returns a ::RESET_REASON_UNKNOWN
58  * value to the user. To avoid this, the user should call the ResetReason API
59  * as early as possible at boot time.
60  * * If the target doesn't support clearing reset registers,
61  * ::hal_reset_reason_get seems to erroneously return a reset reason even after
62  * clearing.
63  *
64  * @{
65  */
66 
67 /** Definitions of different reset reasons
68  */
69 typedef enum {
70  RESET_REASON_POWER_ON, /**< Set when power is initially applied to the board. The power-on-reset circuit causes a POWER_ON reset when this occurs */
71  RESET_REASON_PIN_RESET, /**< Set when a reset is triggered by the hardware pin on the board */
72  RESET_REASON_BROWN_OUT, /**< Triggered when the voltage drops below the low voltage detect (LVD) threshold; the system is held in a reset until the voltage rises above the threshold */
73  RESET_REASON_SOFTWARE, /**< Set during software reset, typically triggered by writing the SYSRESETREQ bit in the Application Interrupt and Reset Control register */
74  RESET_REASON_WATCHDOG, /**< Set when a running watchdog timer fails to be refreshed */
75  RESET_REASON_LOCKUP, /**< Set when the core is locked because of an unrecoverable exception */
76  RESET_REASON_WAKE_LOW_POWER, /**< Set when waking from deep sleep mode */
77  RESET_REASON_ACCESS_ERROR, /**< Umbrella value that encompasses any access related reset */
78  RESET_REASON_BOOT_ERROR, /**< Umbrella value that encompasses any boot related reset */
79  RESET_REASON_MULTIPLE, /**< Set if multiple reset reasons are set within the board. Occurs when the reset reason registers aren't cleared between resets */
80  RESET_REASON_PLATFORM, /**< Platform specific reset reason not captured in this enum */
81  RESET_REASON_UNKNOWN /**< Unknown or unreadable reset reason **/
82 } reset_reason_t;
83 
84 /** Reset reason capabilities of the platform
85  */
86 typedef struct {
87  uint32_t reasons; /**< Supported reset reasons. Each bit represents a corresponding reset_reason_t value.**/
88 } reset_reason_capabilities_t;

90 /** Fetch the reset reason for the last system reset.
91  *
92  * This function must return the contents of the system reset reason registers
93  * cast to an appropriate platform independent reset reason. If multiple reset
94  * reasons are set, this function should return ::RESET_REASON_MULTIPLE. If the
95  * reset reason does not match any existing platform independent value, this
96  * function should return ::RESET_REASON_PLATFORM. If no reset reason can be
97  * determined, this function should return ::RESET_REASON_UNKNOWN.
98  *
99  * This function is not idempotent; there is no guarantee the system
100  * reset reason will not be cleared between calls to this function altering the
101  * return value between calls.
102  *
103  * Note: Some platforms contain reset reason registers that persist through
104  * system resets. If the registers haven't been cleared before calling this
105  * function, multiple reasons may be set within the registers. If multiple reset
106  * reasons are detected, this function returns ::RESET_REASON_MULTIPLE.
107  *
108  * @return enum containing the last reset reason for the board.
109  */
110 reset_reason_t hal_reset_reason_get(void);
111 
112 
113 /** Fetch the raw platform specific reset reason register value.
114  *
115  * This function must return the raw contents of the system reset reason
116  * registers cast to a uint32_t value. If the platform contains reset reasons
117  * that span multiple registers/addresses, the value should be concatenated into
118  * the return type.
119  *
120  * This function is not idempotent; there is no guarantee the system
121  * reset reason will not be cleared between calls to this function altering the
122  * return value between calls.
123  *
124  * @return value containing the reset reason register for the given platform.
125  * If the platform contains reset reasons across multiple registers, they
126  * will be concatenated here.
127  */
128 uint32_t hal_reset_reason_get_raw(void);
129 
130 /** Clear the reset reason from registers.
131  *
132  * Reset the value of the reset status registers. The reset reason persists
133  * between system resets on certain platforms, so the registers should be cleared
134  * before the system resets. Failing to do so may make it difficult to determine
135  * the cause of any subsequent system resets.
136  */
137 void hal_reset_reason_clear(void);
138 
139 /** Fill the given reset_reason_capabilities_t instance according to platform capabilities.
140  */
141 void hal_reset_reason_get_capabilities(reset_reason_capabilities_t *cap);
