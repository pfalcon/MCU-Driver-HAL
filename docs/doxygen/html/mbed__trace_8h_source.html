17 /**
18  * \file mbed_trace.h
19  * Trace interface for MbedOS applications.
20  * This file provide simple but flexible way to handle software traces.
21  * Trace library are abstract layer, which use stdout (printf) by default,
22  * but outputs can be easily redirect to custom function, for example to
23  * store traces to memory or other interfaces.
24  *
25  * usage example:
26  * \code(main.c:)
27  * #include "mbed_trace.h"
28  * #define TRACE_GROUP "main"
29  *
30  * int main(void){
31  * mbed_trace_init(); // initialize trace library
32  * tr_debug("this is debug msg"); //print debug message to stdout: "[DBG]
33  * tr_info("this is info msg");
34  * tr_warn("this is warning msg");
35  * tr_err("this is error msg");
36  * return 0;
37  * }
38  * \endcode
39  * Activate with compiler flag: YOTTA_CFG_MBED_TRACE
40  * Configure trace line buffer size with compiler flag: YOTTA_CFG_MBED_TRACE_LINE_LENGTH. Default length: 1024.
41  * Limit the size of flash by setting MBED_TRACE_MAX_LEVEL value. Default is TRACE_LEVEL_DEBUG (all included)
42  *
43  */

76 /** 3 upper bits are trace modes related,

84 /** plain trace data instead of "headers" */
85 #define TRACE_MODE_PLAIN 0x80
86 /** color mode */

88 /** Use print CR before trace line */
89 #define TRACE_CARRIAGE_RETURN 0x20

91 /** used to activate all trace levels */
92 #define TRACE_ACTIVE_LEVEL_ALL 0x1F
93 /** print all traces same as above */
94 #define TRACE_ACTIVE_LEVEL_DEBUG 0x1f
95 /** print info,warn and error traces */
96 #define TRACE_ACTIVE_LEVEL_INFO 0x0f
97 /** print warn and error traces */
98 #define TRACE_ACTIVE_LEVEL_WARN 0x07
99 /** print only error trace */
100 #define TRACE_ACTIVE_LEVEL_ERROR 0x03
101 /** print only cmd line data */
102 #define TRACE_ACTIVE_LEVEL_CMD 0x01
103 /** trace nothing */
104 #define TRACE_ACTIVE_LEVEL_NONE 0x00
105 
106 /** this print is some deep information for debug purpose */
107 #define TRACE_LEVEL_DEBUG 0x10
108 /** Info print, for general purpose prints */
109 #define TRACE_LEVEL_INFO 0x08
110 /** warning prints, which shouldn't causes any huge problems */
111 #define TRACE_LEVEL_WARN 0x04
112 /** Error prints, which causes probably problems, e.g. out of mem. */
113 #define TRACE_LEVEL_ERROR 0x02
114 /** special level for cmdline. Behaviours like "plain mode" */
115 #define TRACE_LEVEL_CMD 0x01

246 /**
247  * Set trace mutex wait function
248  * By default, trace calls are not thread safe.
249  * If thread safety is required this can be used to set a callback function that will be called before each trace call.
250  * The specific implementation is up to the application developer, but the mutex must count so it can
251  * be acquired from a single thread repeatedly.
252  */
253 void mbed_trace_mutex_wait_function_set(void (*mutex_wait_f)(void));
254 /**
255  * Set trace mutex release function
256  * By default, trace calls are not thread safe.
257  * If thread safety is required this can be used to set a callback function that will be called before returning from
258  * each trace call. The specific implementation is up to the application developer, but the mutex must count so it can
259  * be acquired from a single thread repeatedly.
260  */
261 void mbed_trace_mutex_release_function_set(void (*mutex_release_f)(void));
