45 /**
46  * Memory order constraints for atomic operations. Intended semantics
47  * are as per C++11.
48  */
49 typedef enum mbed_memory_order {
50  /* Bits 0 = consume
51  * 1 = acquire (explicitly requested, or implied by seq.cst)
52  * 2 = release (explicitly requested, or implied by seq.cst)
53  * 4 = sequentially consistent
54  */
55  mbed_memory_order_relaxed = 0x00,
56  mbed_memory_order_consume = 0x01,
57  mbed_memory_order_acquire = 0x02,
58  mbed_memory_order_release = 0x04,
59  mbed_memory_order_acq_rel = 0x06,
60  mbed_memory_order_seq_cst = 0x16
61 } mbed_memory_order;

158 /**
159  * Atomic compare and set. It compares the contents of a memory location to a
160  * given value and, only if they are the same, modifies the contents of that
161  * memory location to a given new value. This is done as a single atomic
162  * operation. The atomicity guarantees that the new value is calculated based on
163  * up-to-date information; if the value had been updated by another thread in
164  * the meantime, the write would fail due to a mismatched expectedCurrentValue.
165  *
166  * Refer to https://en.wikipedia.org/wiki/Compare-and-set [which may redirect
167  * you to the article on compare-and swap].
168  *
169  * @param ptr The target memory location.
170  * @param[in,out] expectedCurrentValue A pointer to some location holding the
171  * expected current value of the data being set atomically.
172  * The computed 'desiredValue' should be a function of this current value.
173  * @note: This is an in-out parameter. In the
174  * failure case of atomic_cas (where the
175  * destination isn't set), the pointee of expectedCurrentValue is
176  * updated with the current value.
177  * @param[in] desiredValue The new value computed based on '*expectedCurrentValue'.
178  *
179  * @return true if the memory location was atomically
180  * updated with the desired value (after verifying
181  * that it contained the expectedCurrentValue),
182  * false otherwise. In the failure case,
183  * exepctedCurrentValue is updated with the new
184  * value of the target memory location.
185  *
186  * pseudocode:
187  * function cas(p : pointer to int, old : pointer to int, new : int) returns bool {
188  * if *p != *old {
189  * *old = *p
190  * return false
191  * }
192  * *p = new
193  * return true
194  * }
195  *
196  * @note: In the failure case (where the destination isn't set), the value
197  * pointed to by expectedCurrentValue is instead updated with the current value.
198  * This property helps writing concise code for the following incr:
199  *
200  * function incr(p : pointer to int, a : int) returns int {
201  * done = false
202  * value = atomic_load(p)
203  * while not done {
204  * done = atomic_cas(p, &value, value + a) // *value gets updated automatically until success
205  * }
206  * return value + a
207  * }
208  *
209  * However, if the call is made in a loop like this, the atomic_compare_exchange_weak
210  * functions are to be preferred.
211  *
212  * @note: This corresponds to the C11 "atomic_compare_exchange_strong" - it
213  * always succeeds if the current value is expected, as per the pseudocode
214  * above; it will not spuriously fail as "atomic_compare_exchange_weak" may.
215  * This call would normally be used when a fail return does not retry.
216  */
217 MBED_INLINE_IF_EX bool core_util_atomic_cas_u8(volatile uint8_t *ptr, uint8_t *expectedCurrentValue, uint8_t desiredValue);

281 /**
282  * Atomic compare and set. It compares the contents of a memory location to a
283  * given value and, only if they are the same, modifies the contents of that
284  * memory location to a given new value. This is done as a single atomic
285  * operation. The atomicity guarantees that the new value is calculated based on
286  * up-to-date information; if the value had been updated by another thread in
287  * the meantime, the write would fail due to a mismatched expectedCurrentValue.
288  *
289  * Refer to https://en.wikipedia.org/wiki/Compare-and-set [which may redirect
290  * you to the article on compare-and swap].
291  *
292  * @param ptr The target memory location.
293  * @param[in,out] expectedCurrentValue A pointer to some location holding the
294  * expected current value of the data being set atomically.
295  * The computed 'desiredValue' should be a function of this current value.
296  * @note: This is an in-out parameter. In the
297  * failure case of atomic_cas (where the
298  * destination isn't set), the pointee of expectedCurrentValue is
299  * updated with the current value.
300  * @param[in] desiredValue The new value computed based on '*expectedCurrentValue'.
301  *
302  * @return true if the memory location was atomically
303  * updated with the desired value (after verifying
304  * that it contained the expectedCurrentValue),
305  * false otherwise. In the failure case,
306  * exepctedCurrentValue is updated with the new
307  * value of the target memory location.
308  *
309  * pseudocode:
310  * function cas(p : pointer to int, old : pointer to int, new : int) returns bool {
311  * if *p != *old or spurious failure {
312  * *old = *p
313  * return false
314  * }
315  * *p = new
316  * return true
317  * }
318  *
319  * @note: In the failure case (where the destination isn't set), the value
320  * pointed to by expectedCurrentValue is instead updated with the current value.
321  * This property helps writing concise code for the following incr:
322  *
323  * function incr(p : pointer to int, a : int) returns int {
324  * done = false
325  * value = *p // This fetch operation need not be atomic.
326  * while not done {
327  * done = atomic_compare_exchange_weak(p, &value, value + a) // *value gets updated automatically until success
328  * }
329  * return value + a
330  * }
331  *
332  * @note: This corresponds to the C11 "atomic_compare_exchange_weak" - it
333  * may spuriously fail if the current value is expected, as per the pseudocode
334  * above; it will not spuriously fail as "atomic_compare_exchange_weak" may.
335  * This call would normally be used when a fail return will cause a retry anyway,
336  * saving the need for an extra loop inside the cas operation.
337  */
